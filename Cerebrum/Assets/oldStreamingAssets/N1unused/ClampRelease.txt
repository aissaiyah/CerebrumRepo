Player lookat $InfusionClamp
ZoomSpot Menu.Question 'Release Clamp?'
ZoomSpot Menu.Choices 'Yes','No'
ZoomSpot Menu.on ZoomSpot
WaitFor
	ZoomSpot Menu.Done
Wait 1
If
	ZoomSpot menu.result $Yes
Then
	$ClampOpen = true
	Prompt 'The Clamp has been released.'
	Wait 1
	If 
		$IVConnected
	Then
		ScoreTracker rangecomplete 0-14
		Do 'EndCheck.txt'
	Endif

	If
		$ChamberAmount > 0
	Then
		turnOn $IVLineBlood
		Prompt 'Blood is flowing'
		Wait 1
	Endif

	Prompt ''
	#ScoreTracker gain 1
	If
		ScoreTracker iscomplete 10
	Then
		ScoreTracker taskCorrect 13 true
	Else
		ScoreTracker taskCorrect 10 true
	Endif
#	Do '06/ClampClose.txt'
Else
	If
		ScoreTracker iscomplete 10
	Then
		ScoreTracker taskCorrect 13 false
	Else
		ScoreTracker taskCorrect 10 false
	Endif
	Return
	#Do '06/InfusionStart.txt'
Endif
