/StartCanvas off
/JoystickCanvas off
/PatientCanvas on
#Player lookat $PlayerPatLook
$Patient outline.on
$Patient clickable true 
Prompt 'Patient = ' + Patient
Wait 5
Prompt ''

Label 'P01Loop'
	If
		PatientCanvas/Choices/Speak pressed
	Then
		PatientCanvas/Choices/Speak reset

		prompt 'Good Job'
	Endif
	If
		PatientCanvas/Choices/Leave pressed
	Then
		PatientCanvas/Choices/Leave reset
		$Patient reset
		/PatientCanvas off
		/JoystickCanvas on
		/StartCanvas on
		/Player align PlayerStart
		Return
	Endif
	Wait .1
Goto 'P01Loop'
